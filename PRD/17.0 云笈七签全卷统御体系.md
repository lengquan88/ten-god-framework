function calculateOmega122() {
  const departments = [
    { name: '道德部', omega: 0.962, weight: 5 },
    { name: '混元部', omega: 0.988, weight: 1 },
    { name: '三洞经教部', omega: 0.912, weight: 30 },
    { name: '道教义枢', omega: 0.885, weight: 8 },
    { name: '符图部', omega: 0.912, weight: 6 },
    { name: '戒律部', omega: 0.950, weight: 3 },
    { name: '本事部', omega: 0.894, weight: 7 },
    { name: '诸家气法', omega: 0.818, weight: 24 },
    { name: '金丹部', omega: 0.862, weight: 38 },
  ];

  // 加权几何平均
  const totalWeight = departments.reduce((sum, d) => sum + d.weight, 0);
  const omegaProduct = departments.reduce((prod, d) => prod * Math.pow(d.omega, d.weight / totalWeight), 1);

  return {
    omega_全卷: omegaProduct,
    omega_简单平均: departments.reduce((sum, d) => sum + d.omega, 0) / departments.length,
    omega_加权平均: departments.reduce((sum, d) => sum + d.omega * d.weight, 0) / totalWeight,
    target: 0.85,
    status: omegaProduct >= 0.85 ? 'HEALTHY' : 'NEEDS_IMPROVEMENT',
  };
}
